29 polynomialgf<P> derivative(const polynomialgf<P> &val) {
30  polynomialgf<P> res = val;
31  auto i = val.degree();
32  for (res[i] = 0; i > 0; --i) {
33  res[i - 1] = gf<P>(i) * val[i];
34  }
35  res.normalize();
36  return res;

44 template<uint32_t P>
45 polynomialgf<P> gcd(polynomialgf<P> m, polynomialgf<P> n) {
46  if (m.is_zero() || n.is_zero()) {
47  throw std::domain_error("arguments must be strictly positive");
48  }
49  if (m.degree() < n.degree()) {
50  std::swap(m, n);
51  }
52  polynomialgf<P> u0 = m, u1 = polynomialgf<P>({1}), u2 = polynomialgf<P>({0}),
53  v0 = n, v1 = polynomialgf<P>({0}), v2 = polynomialgf<P>({1}),
54  w0, w1, w2, q;
55  while (!v0.is_zero()) {
56  q = u0 / v0;
57  w0 = u0 - q * v0, w1 = u1 - q * v1, w2 = u2 - q * v2;
58  u0 = v0, u1 = v1, u2 = v2, v0 = w0, v1 = w1, v2 = w2;
59  }
60  return u0;
61 }

79  template<uint32_t P>
80  [[nodiscard]]
81  polynomialgf<P> x_pow_mod_sub(
82  uint64_t pow,
83  const polynomialgf<P> &mod,
84  const polynomialgf<P> &sub = polynomialgf<P>({0})
85  ) noexcept {
86  // Эта функция возврадает по сути следующий редультат:
87  // return ((polynomialgf<P>({1}) << pow) - sub) % mod;
88  // её существование необходимо, т.к. в методе Рабина
89  // и в методе проверки на примитивность требуется вычислить
90  // её результат для столь больших pow, что x^pow не влезает
91  // в оперативную память, данный метод решает эту проблему,
92  // но он всё ещё адски медленный
93  const auto n = mod.degree();
94  polynomialgf<P> res({1});
95 
96  typename polynomialgf<P>::size_type tmp;
97  for (auto m = res.degree(); pow + m >= n; m = res.degree()) {
98  tmp = n - m;
99  res <<= tmp;
100  res %= mod;
101  pow -= tmp;
102  }
103  return (res << pow) - sub;
104  }
105 }

128 template<uint32_t P>
129 bool is_irreducible_berlekamp(const polynomialgf<P> &val) {
130  if (val.is_zero()) {
131  return false; }
132  const auto n = val.degree();
133 
134  // проверка вырожденных случаев
135  if (n == 0 || (val[0].is_zero() && n > 1)) { return false; }
136  if (n == 1) { return true; }
137 
138  // функция для построения матрицы берлекампа и вычисления её ранга
139  auto berlekampMatrixRank = [](const polynomialgf<P> &val) {
140  polynomialgf<P> tmp;
141  typename polynomialgf<P>::size_type i, j, k, l;
142  const gf<P> zer = 0;
143  const auto n = val.degree();
144  std::vector<std::vector<gf<P>>> m(n, std::vector<gf<P>>(n, zer)); // M = 0
145  for (i = 0; i < n; ++i) {
146  tmp = detail::x_pow_mod_sub(i * P, val); // M[i,*] = x ^ ip (mod val)
147  for (j = 0, k = tmp.degree(); j <= k; ++j) {
148  m[i][j] += tmp[j];
149  }
150  m[i][i] -= 1; // M - E
151  }
152 
153  // приведение матрицы к ступенчатому виду
154  bool f;
155  gf<P> mul;
156  for (i = k = 0; i < n && k < n; ++k) {
157  f = !m[i][k].is_zero();
158  for (j = i + 1; j < n; ++j) {
159  if (!m[j][k].is_zero()) {
160  if (f) {
161  mul = m[i][k].mul_inv() * m[j][k];
162  m[j][k] = zer;
163  for (l = k + 1; l < n; ++l) {
164  m[j][l] -= m[i][l] * mul;
165  }
166  } else {
167  for (l = k; l < n; ++l) {
168  std::swap(m[i][l], m[j][l]);
169  }
170  f = true;
171  }
172  }
173  }
174  i += f;
175  }
176  return i;
177  };
178 
179  // алгоритм Берлекампа
180  auto d = derivative(val);
181  return !d.is_zero() && gcd(val, d).degree() == 0 &&
182  berlekampMatrixRank(val) == val.degree() - 1;
183 }

206 template<uint32_t P>
207 bool is_primitive_definition(const polynomialgf<P> &val) {
208  if (val.is_zero()) { return false; }
209  const auto n = val.degree();
210 
211  // проверка вырожденных случаев
212  if (n == 0 || (val[0].is_zero() && n > 1)) { return false; }
213  if (n == 1 && val[0] == 0) { return true; } // val = k * x + 0
214 
215  // выполняется нормировка, т.к. данный алгоритм справедлив только
216  // для многочленов со старшим коэффициентом, равным единице
217  // умножение многочлена на число не меняет его примитивность
218  const auto poly = val / val[n];
219 
220  // ещё один вырожденный случай, на работу с которым алгоритм не рассчитан
221  if (P == 2 && poly == polynomialgf<P>({1, 1})) { return false; }
222 
223  auto mp = (n % 2) ? -poly[0] : poly[0];
224 
225  // функция для разложения (факторизации) числа на множители
226  // единица и само число (в случае его простоты) в разложение не входят
227  auto factorize = [](uint64_t n) {
228  std::vector<uint64_t> list;
229  const auto begin = n;
230  for (uint64_t d = 2; d * d <= n; ++d) {
231  if (n % d) { continue; }
232  list.emplace_back(d);
233  while (n % d == 0) { n /= d; }
234  }
235  if (n != 1 && n != begin) { list.emplace_back(n); }
236  return list;
237  };
238 
239  // проверяется выполнение первого условия
240  if (P > 2) {
241  const auto p = P - 1;
242  auto list = (p == 2) ? std::vector<uint64_t>{2} : factorize(p);
243  auto m = list.size() - 1;
244  auto tmp = mp;
245  for (uint32_t i = 1; i <= p; ++i, tmp *= mp) {
246  if (i != p / list[m]) { continue; }
247  if (tmp.data() == 1) { return false; }
248  if (m == 0) { break; } else { m -= 1; }
249  }
250  }
251 
252  // проверяется выполнение второго условия
253  uint64_t r = (detail::integer_power(static_cast<uint64_t>(P), n) - 1) / (P - 1);
254  auto tmp = detail::x_pow_mod_sub(r, val, polynomialgf<P>({mp}));
255  if (!tmp.is_zero()) { return false; }
256 
257  // проверяется выполнение третьего условия
258  auto list3 = factorize(r);
259  const auto m = list3.size();
260  for (size_t i = 0; i < m; ++i) {
261  tmp = detail::x_pow_mod_sub(r / list3[i], poly);
262  if (tmp.is_zero() || tmp.degree() == 0) { return false; }
263  }
264 
265  // если все условия выполнены - многочлен примитивен
266  return true;
267 }

285  if (val.is_zero()) { return false; }

292  // функция разложения числа на множители
293  auto get_list = [](uint64_t n) {
294  std::vector<uint64_t> list;
295  const auto begin = n;
296  for (uint64_t d = 2; d * d <= n; ++d) {
297  if (n % d) { continue; }
298  list.emplace_back(begin / d);
299  while (n % d == 0) { n /= d; }
300  }
301  if (n != 1) { list.emplace_back(begin / n); }
302  return list;
303  };
304 
305  // шаги 1-2
306  auto list = get_list(n);
307  polynomialgf<P> tmp, x = polynomialgf<P>({0, 1});
308  for (auto i: list) {
309  tmp = detail::x_pow_mod_sub(detail::integer_power(static_cast<uint64_t>(P), i), val, x);
310  if (tmp.is_zero() || gcd(val, tmp).degree() > 0) { return false; }
311  }
312 
313  // шаг 3
314  tmp = detail::x_pow_mod_sub(detail::integer_power(static_cast<uint64_t>(P), n), val, x);
315  return tmp.is_zero();
316 }
