27 auto gcd(gfpoly m, gfpoly n) -> gfpoly {
28  assert(m.field() == n.field());
29  if (m.is_zero() || n.is_zero()) {
30  throw std::domain_error("arguments must be strictly positive");
31  }
32  if (m.degree() < n.degree()) {
33  std::swap(m, n);
34  }
35  gfpoly u0 = m, u1 = gfpoly(m.field(), 1), u2 = gfpoly(m.field()),
36  v0 = n, v1 = gfpoly(m.field()), v2 = gfpoly(m.field(), 1),
37  w0 = gfpoly(m.field()), w1 = gfpoly(m.field()), w2 = gfpoly(m.field()),
38  q = gfpoly(m.field());
39  while (!v0.is_zero()) {
40  q = u0 / v0;
41  w0 = u0 - q * v0, w1 = u1 - q * v1, w2 = u2 - q * v2;
42  u0 = v0, u1 = v1, u2 = v2, v0 = w0, v1 = w1, v2 = w2;
43  }
44  return u0;
45 }

51 auto integer_power(T t, N n) -> T {
52  switch (n) {
53  case 0:return static_cast<T>(1U);
54  case 1:return t;
55  case 2:return t * t;
56  case 3:return t * t * t;
57  default:T result = integer_power(t, n / 2);
58  return (n & 1U) ? result * result * t : result * result;
59  }
60 }

80 [[nodiscard]]
81 auto x_pow_mod(uintmax_t pow, const gfpoly &mod) -> gfpoly {
82  const auto n = mod.degree();
83  gfpoly xn = gfpoly(mod.field(), 1) << n; // x^n
84  gfpoly res(mod.field(), 1);
85 
86  uintmax_t d = 0;
87  uintmax_t tmp = 0;
88  for (auto m = res.degree(); pow + m >= n; m = res.degree()) {
89  tmp = n - m, pow -= tmp, res <<= tmp;
90  d = (res == xn) ? (d) ? (d -= pow, pow %= d, 0) : pow : d;
91  res %= mod;
92  }
93  return res << pow;
94 }

132  // функция для построения матрицы берлекампа и вычисления её ранга
133  auto berlekampMatrixRank = [](const gfpoly &val) {
134  gfpoly poly(val.field());
135  uintmax_t i = 0, j = 0, k = 0, l = 0;
136  const auto n = val.degree();
137  std::vector<std::vector<gfn>> m(n, std::vector<gfn>(n, gfn(val.field()))); // M = 0
138  for (i = 0; i < n; ++i) {
139  // M[i,*] = x ^ ip (mod val)
140  poly = detail::x_pow_mod(i * val.base(), val);
141  for (j = 0, k = poly.degree(); j <= k; ++j) {
142  m[i][j] += poly[j];
143  }
144  m[i][i] -= 1; // M - E
145  }
146 
147  // приведение матрицы к ступенчатому виду
148  bool f = false;
149  gfn num(val.field());
150  for (i = k = 0; i < n && k < n; ++k) {
151  f = !m[i][k].is_zero();
152  for (j = i + 1; j < n; ++j) {
153  if (!m[j][k].is_zero()) {
154  if (f) {
155  num = m[j][k] / m[i][k];
156  m[j][k].set_zero();
157  for (l = k + 1; l < n; ++l) {
158  m[j][l] -= m[i][l] * num;
159  }
160  } else {
161  for (l = k; l < n; ++l) {
162  std::swap(m[i][l], m[j][l]);
163  }
164  f = true;
165  }
166  }
167  }
168  i += f;
169  }
170  return i;
171  };
172 
173  // алгоритм Берлекампа
174  auto d = detail::derivative(val);
175  return !d.is_zero() && gcd(val, d).degree() == 0 &&
176  berlekampMatrixRank(val) == val.degree() - 1;
177 }

293 auto is_primitive_definition(const gfpoly &val) -> bool {
294  if (val.is_zero()) {
295  return false;
296  }
297  const auto n = val.degree();
298 
299  // проверка вырожденных случаев
300  if (n == 0 || (val[0].is_zero() && n > 1)) {
301  return false;
302  }
303  if (n == 1 && val[0] == 0) {
304  return true;
305  } // val = k * x + 0
306 
307  // выполняется нормировка, т.к. данный алгоритм справедлив только
308  // для многочленов со старшим коэффициентом, равным единице
309  // умножение многочлена на число не меняет его примитивность
310  const auto poly = val / val[n];
311 
312  // ещё один вырожденный случай, на работу с которым алгоритм не рассчитан
313  auto P = poly.base();
314  if (P == 2 && poly == gfpoly(poly.field(), {1, 1})) {
315  return false;
316  }
317 
318  gfn mp = (n % 2) ? -poly[0] : poly[0];
319 
320  // функция для разложения (факторизации) числа на множители
321  // единица и само число (в случае его простоты) в разложение не входят
322  auto factorize = [](uintmax_t n) {
323  std::vector<uintmax_t> list;
324  const auto begin = n;
325  for (uintmax_t d = 2; d * d <= n; ++d) {
326  if (n % d) {
327  continue;
328  }
329  list.emplace_back(d);
330  while (n % d == 0) {
331  n /= d;
332  }
333  }
334  if (n != 1 && n != begin) {
335  list.emplace_back(n);
336  }
337  return list;
338  };
339 
340  // проверяется выполнение первого условия
341  if (P > 2) {
342  const auto p = P - 1;
343  auto list = (p == 2) ? std::vector<uintmax_t>{2} : factorize(p);
344  auto m = list.size() - 1;
345  auto tmp = mp;
346  for (uint32_t i = 1; i <= p; ++i, tmp *= mp) {
347  if (i != p / list[m]) {
348  continue;
349  }
350  if (tmp == 1) {
351  return false;
352  }
353  if (m == 0) {
354  break;
355  }
356  m -= 1;
357  }
358  }
359 
360  // проверяется выполнение второго условия
361  uintmax_t r = (detail::integer_power(P, n) - 1) / (P - 1);
362  auto tmp = detail::x_pow_mod(r, val) - mp;
363  if (!tmp.is_zero()) {
364  return false;
365  }
366 
367  // проверяется выполнение третьего условия
368  auto list3 = factorize(r);
369  const auto m = list3.size();
370  for (size_t i = 0; i < m; ++i) {
371  tmp = detail::x_pow_mod(r / list3[i], poly);
372  if (tmp.is_zero() || tmp.degree() == 0) {
373  return false;
374  }
375  }
376 
377  // если все условия выполнены - многочлен примитивен
378  return true;
379 }
