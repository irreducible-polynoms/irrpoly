257  void check(uint64_t n, input_func in, check_func cf, callback_func back, const bool strict = true) noexcept {
258  // заряжаем многочлены на проверку
259  for (auto *sl : s) {
260  sl->set_check(cf);
261  sl->set(in());
262  }
263  while (n) {
264  // ждём свободный поток
265  while (countBusy() == s.size()) { m.wait(); }
266  // находим свободные потоки и заряжаем новыми входными данными
267  for (unsigned i = 0; i < s.size() && n; ++i) {
268  if (s[i]->busy()) { continue; }
269  if (back(s[i]->get(), s[i]->result())) { --n; }

272  }

274  while (countBusy()) { m.wait(); }

277  for (auto *sl : s) { back(sl->get(), sl->result()); }
278  }
