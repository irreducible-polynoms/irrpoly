292  void check(uint64_t n, input_func input, callback_func callback, const bool strict = true) noexcept {
293  // заряжаем многочлены на проверку
294  for (auto *sl : s) { sl->set(input()); }
295  while (n) {
296  // ждём свободный поток
297  while (countBusy() == s.size()) { m.wait(); }
298  // находим свободные потоки и заряжаем новыми входными данными
299  for (unsigned i = 0; i < s.size() && n; ++i) {
300  if (s[i]->busy()) { continue; }
301  if (callback(s[i]->result(), s[i]->get())) { --n; }
302  s[i]->set(input());

304  }
305  // ожидаем завершения всех потоков
306  while (countBusy()) { m.wait(); }
307  if (!strict) {
308  // обрабатываем все проверенные многочлены, даже если их больше, чем требовалось найти
309  for (auto *sl : s) { callback(sl->result(), sl->get()); }
310  }
