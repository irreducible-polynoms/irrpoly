165  void check(uint64_t n, input_func in, check_func cf, callback_func back, const bool strict = true) {
166  std::unique_lock<std::mutex> lk(*s_mutex);
167  // заряжаем многочлены на проверку
168  for (const auto &sl : s) {
169  sl->set_check(cf);
170  sl->set(in());
171  }
172  while (n) {
173  // ждём свободный поток
174  while (countBusy() == s.size()) {
175  s_cond->wait(lk);
176  }
177  // находим свободные потоки и заряжаем новыми входными данными
178  for (unsigned i = 0; i < s.size() && n; ++i) {
179  if (s[i]->busy()) {
180  continue;
181  }
182  if (back(s[i]->get().value(), s[i]->result().value())) {
183  --n;
184  }
185  s[i]->set(in());
186  }
187  }
188  // ожидаем завершения всех потоков
189  while (countBusy()) {
190  s_cond->wait(lk);
191  }
192  if (!strict) {
193  // обрабатываем все проверенные многочлены, даже если их больше, чем требовалось найти
194  for (const auto &sl : s) {
195  back(sl->get().value(), sl->result().value());
196  }
197  }
198  }
