255  void check(uint64_t n, input_func in, check_func cf, callback_func back, const bool strict = true) noexcept {
256  // заряжаем многочлены на проверку
257  for (auto *sl : s) {

259  sl->set(in());
260  }
261  while (n) {
262  // ждём свободный поток
263  while (countBusy() == s.size()) { m.wait(); }
264  // находим свободные потоки и заряжаем новыми входными данными
265  for (unsigned i = 0; i < s.size() && n; ++i) {
266  if (s[i]->busy()) { continue; }
267  if (back(s[i]->get(), s[i]->result())) { --n; }
